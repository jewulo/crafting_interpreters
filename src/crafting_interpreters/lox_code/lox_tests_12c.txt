class Person {
    sayName() {
        print this.name;
    }
}

var jane = Person();
jane.name = "Jane";
jane.sayName();

var method = jane.sayName;
method();

var bill = Person();
bill.name = "Bill";
bill.sayName();

bill.sayName = jane.sayName;
bill.sayName();

class Egotist {
    speak() {
        print this;
    }
}

var method = Egotist().speak;
method();

class Cake {
    taste() {
        var adjective = "delicious";
        print "The " + this.flavour + " cake is " + adjective + "!";
    }
}

var cake = Cake();
cake.flavour = "German chocolate";
cake.taste();

class Thing {
    getCallBack() {
        fun localFunction() {
            print this;
        }

        return localFunction;
    }
}

var callback = Thing().getCallBack();
callback();
